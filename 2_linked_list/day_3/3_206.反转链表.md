**2.第二次写，遇到的问题**
```python
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next
class Solution:    
    def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:  # 添加边界检查
            return head

        dummy_head=ListNode(next=head)
        
        pre2=None
        current=dummy_head.next
        pre1=current.next

        count=1     
        while pre1:
            if count==1:      #! 1->2   2->1  有一个环，要去掉
                current.next = pre2
                count-=1
            
            dummy_head.next=pre1
            pre2=pre1.next
            pre1.next=current
            current=pre1
            pre1=pre2

        return dummy_head.next
```
